#!/bin/bash

# Update packages and install necessaries softwares
yum update -y
yum install docker git -y

sleep 60

# Docker group
sudo usermod -aG docker ec2-user

# Enable and Start Aplications
sudo systemctl start docker
sudo systemctl enable docker

# Cloning app
cd /home/ec2-user
git clone https://github.com/LucasMorgani/teste-docker.git
sleep 20
cd teste-docker

# Starting docker
docker compose -f Compose.yml up -d
