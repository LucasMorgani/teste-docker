#!/bin/bash

# Update packages and install necessaries softwares
yum update -y
yum install docker git -y

sleep 60
# Docker group
sudo usermod -aG docker ec2-user

# Enable and Start Aplications
sudo systemctl start docker
sudo systemctl enable docker

# Install Docker Compose
DOCKER_CONFIG=/root/.docker
mkdir -p $DOCKER_CONFIG/cli-plugins
curl -SL https://github.com/docker/compose/releases/download/v2.24.5/docker-compose-linux-x86_64 -o $DOCKER_CONFIG/cli-plugins/docker-compose
chmod +x $DOCKER_CONFIG/cli-plugins/docker-compose

# Cloning app
cd /root
git clone https://github.com/LucasMorgani/teste-docker.git
sleep 20
cd /home/ec2-user/teste-docker

# Starting docker
/root/.docker/cli-plugins/docker-compose -f Compose.yml up -d
